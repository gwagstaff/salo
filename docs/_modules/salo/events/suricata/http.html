[docs]    @validator("http_version", pre=True, always=True)
    def set_http_version(cls, v):
        if v:
            if not v.startswith("HTTP/"):
                v = "HTTP/" + v
        return v or random.choice(HTTP_VERSIONS)



[docs]    @validator("http_status_code", pre=True, always=True)
